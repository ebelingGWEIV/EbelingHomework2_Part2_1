/*

 * 
 */
int testPalindrome(const char *myString)
{
    return Palindrome(myString, strlen(myString), 0);
}

// r    a   d   a   r
int Palindrome(const char *myString, int length, int n)
{
    if(n == strlen(myString) / 2)
        return 1; //If the middle character is reached without failing, it's a palindrome
    if(myString[n] == myString[n])
        return Palindrome(myString, length-1, ++n); //If the first an last letters match, then check the next pair
    else
        return 0; //If there is no match, it wasn't a palindrome.

}
